# Create the environment if ot is not created
python3 -m venv venv

# Activate it this is the next step to start allways the server
source venv/bin/activate

# Install Flask if it is not intalled
pip install flask

# Install pyOpenSSL to support HTTPS 'adhoc' certificates
pip install pyopenssl

# Install other necessary dependencies for production
pip install flask-cors mysql-connector-python bcrypt python-dotenv requests gunicorn
